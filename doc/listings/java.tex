\usepackage{xcolor}
\usepackage{listings}

% Definizione colori
\definecolor{keywordcolor}{RGB}{0,0,150}       % blu scuro per keyword
\definecolor{commentcolor}{RGB}{0,128,0}       % verde per commenti
\definecolor{stringcolor}{RGB}{163,21,21}      % rosso scuro per stringhe
\definecolor{background}{HTML}{F7F7F7}         % grigio chiarissimo sfondo
\definecolor{numbercolor}{RGB}{100,0,100}      % viola per numeri

% Definizione linguaggio java per listings
\lstdefinelanguage{java}{
	keywords=[1]{abstract,assert,boolean,break,byte,case,catch,char,class,const,continue,
		default,do,double,else,enum,extends,false,final,finally,float,for,goto,if,implements,
		import,instanceof,int,interface,long,native,new,null,package,private,protected,public,
		return,short,static,strictfp,super,switch,synchronized,this,throw,throws,transient,true,
		try,void,volatile,while},
	keywordstyle=\color{keywordcolor}\bfseries,
	ndkeywords={@Document,Id}, % annotazioni
	ndkeywordstyle=\color{blue}\bfseries,
	sensitive=true,
	comment=[l]{//},
	morecomment=[s]{/*}{*/},
	commentstyle=\color{commentcolor}\itshape,
	stringstyle=\color{stringcolor},
	morestring=[b]",
	morestring=[b]',
	numberstyle=\color{numbercolor},
	basicstyle=\footnotesize\ttfamily,
	numbers=left,
	numbersep=8pt,
	stepnumber=1,
	backgroundcolor=\color{background},
	showstringspaces=false,
	breaklines=true,
	frame=lines,
	captionpos=b,
	tabsize=2,
	escapeinside={(*@}{@*)}
}
